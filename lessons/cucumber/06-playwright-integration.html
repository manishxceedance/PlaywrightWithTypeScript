Lesson 6: Playwright Integration — Real Browser Automation

Level: Intermediate Wire Playwright's browser API into your Cucumber step definitions end-to-end.

1. How Playwright Fits Into the Stack

In this architecture, Cucumber is the test runner and Playwright is the browser automation engine. Cucumber reads feature files, routes steps to TypeScript functions, and those functions call Playwright to interact with a real browser. 

Feature File (.feature)
  → Cucumber Runner (cucumber-js)
     → Step Definitions (.ts)
        → Playwright API (page.click, page.fill, expect)
           → Real Browser (Chromium / Firefox / WebKit)

2. Complete World Setup

The World object is the bridge. It holds the Playwright browser, context, and page references that every step definition can access through this: 

// src/support/world.ts
import { setWorldConstructor, World, IWorldOptions } from '@cucumber/cucumber';
import { Browser, BrowserContext, Page, chromium } from '@playwright/test';

export interface CustomWorld extends World {
  browser: Browser;
  context: BrowserContext;
  page: Page;
  baseUrl: string;
}

class PlaywrightWorld extends World {
  browser!: Browser;
  context!: BrowserContext;
  page!: Page;
  baseUrl: string = process.env.BASE_URL ?? 'https://manishxceedance.github.io/PlaywrightWithTypeScript/lessons/practice-app/';

  constructor(options: IWorldOptions) {
    super(options);
  }
}

setWorldConstructor(PlaywrightWorld);

3. Complete Hooks Setup

// src/support/hooks.ts
import { Before, After, BeforeAll, AfterAll, Status } from '@cucumber/cucumber';
import { chromium, Browser } from '@playwright/test';
import { CustomWorld } from './world';

let browser: Browser;

BeforeAll(async function () {
  browser = await chromium.launch({
    headless: process.env.HEADED !== 'true',
    slowMo: process.env.SLOW_MO ? parseInt(process.env.SLOW_MO) : 0
  });
});

Before(async function (this: CustomWorld) {
  this.context = await browser.newContext({
    viewport: { width: 1280, height: 720 },
    ignoreHTTPSErrors: true
  });
  this.page = await this.context.newPage();
});

After(async function (this: CustomWorld, scenario) {
  if (scenario.result?.status === Status.FAILED) {
    const screenshot = await this.page.screenshot({ fullPage: true });
    this.attach(screenshot, 'image/png');

    const html = await this.page.content();
    this.attach(html, 'text/html');
  }
  await this.context.close();
});

AfterAll(async function () {
  await browser.close();
});

4. Writing Full Step Definitions with Playwright

Here is a complete, working example covering navigation, form interaction, and assertions for a login feature: 

// src/features/login.feature
Feature: Login

  @smoke
  Scenario: Successful login
    Given the user is on the login page
    When the user enters email "admin@demo.com" and password "Admin123!"
    And the user submits the login form
    Then the dashboard heading should be visible
    And the URL should contain "dashboard"
// src/steps/login.steps.ts
import { Given, When, Then } from '@cucumber/cucumber';
import { expect } from '@playwright/test';
import { CustomWorld } from '../support/world';

Given('the user is on the login page', async function (this: CustomWorld) {
  await this.page.goto(this.baseUrl + '/login');
  await expect(this.page.locator('h1')).toContainText('Login');
});

When(
  'the user enters email {string} and password {string}',
  async function (this: CustomWorld, email: string, password: string) {
    await this.page.fill('[data-testid="email"]', email);
    await this.page.fill('[data-testid="password"]', password);
  }
);

When('the user submits the login form', async function (this: CustomWorld) {
  await this.page.click('[data-testid="login-submit"]');
});

Then('the dashboard heading should be visible', async function (this: CustomWorld) {
  await expect(this.page.locator('h1.dashboard-title')).toBeVisible();
});

Then('the URL should contain {string}', async function (this: CustomWorld, path: string) {
  await expect(this.page).toHaveURL(new RegExp(path));
});

5. Playwright Assertions Reference for BDD

These are the most commonly used assertions in Then steps:

// Element is visible
await expect(this.page.locator('.success-toast')).toBeVisible();

// Element has specific text
await expect(this.page.locator('h1')).toHaveText('Welcome back');

// Element contains partial text
await expect(this.page.locator('.message')).toContainText('saved');

// URL matches pattern
await expect(this.page).toHaveURL(/dashboard/);

// Input has a value
await expect(this.page.locator('#email')).toHaveValue('test@test.com');

// Element count
await expect(this.page.locator('.product-card')).toHaveCount(12);

// Element is NOT visible
await expect(this.page.locator('.error')).not.toBeVisible();

// Checkbox is checked
await expect(this.page.locator('#terms')).toBeChecked();

// Element has attribute
await expect(this.page.locator('button[type="submit"]')).toBeEnabled();

6. Handling Waits Properly

Playwright automatically waits for elements to be visible before interacting with them. Do not add manual waits. However, for dynamic content, use waitFor: 

// Wait for a dynamic element to appear
await this.page.locator('.loading-spinner').waitFor({ state: 'hidden' });

// Wait for navigation to complete
await this.page.waitForURL(/confirmation/);

// Wait for an element to be enabled
await this.page.locator('#submit').waitFor({ state: 'visible' });

// Wait for network request to complete
await this.page.waitForResponse(resp => resp.url().includes('/api/orders'));

7. Multi-Step Form Example

# src/features/registration.feature
Feature: User Registration

  Scenario: Complete registration flow
    Given the user opens the registration page
    When the user fills in the registration form:
      | Field      | Value               |
      | First Name | Alice               |
      | Last Name  | Smith               |
      | Email      | alice@hrms.local    |
      | Password   | StrongPass1!        |
    And the user accepts the terms and conditions
    And the user clicks the Register button
    Then the registration success message should appear
    And the user should be redirected to the login page
// src/steps/registration.steps.ts
import { Given, When, Then } from '@cucumber/cucumber';
import { DataTable } from '@cucumber/cucumber';
import { expect } from '@playwright/test';
import { CustomWorld } from '../support/world';

Given('the user opens the registration page', async function (this: CustomWorld) {
  await this.page.goto(this.baseUrl + 'hrms/register.html');
});

When('the user fills in the registration form:',
  async function (this: CustomWorld, table: DataTable) {
    const data = table.rowsHash();
    const fieldMap: Record<string, string> = {
      'First Name': '[name="firstName"]',
      'Last Name': '[name="lastName"]',
      'Email': '[name="email"]',
      'Password': '[name="password"]'
    };

    for (const [label, selector] of Object.entries(fieldMap)) {
      if (data[label]) {
        await this.page.fill(selector, data[label]);
      }
    }
  }
);

When('the user accepts the terms and conditions', async function (this: CustomWorld) {
  await this.page.check('[data-testid="agree-terms"]');
});

When('the user clicks the Register button', async function (this: CustomWorld) {
  await this.page.click('[data-testid="btn-submit"]');
});

Then('the registration success message should appear', async function (this: CustomWorld) {
  await expect(this.page.locator('.toast-success')).toBeVisible();
});

Then('the user should be redirected to the login page', async function (this: CustomWorld) {
  await expect(this.page).toHaveURL(/login/);
});

8. Environment Variables for Configuration

Use environment variables to avoid hardcoding URLs and credentials in test files: 

// .env (never commit to source control)
BASE_URL=https://manishxceedance.github.io/PlaywrightWithTypeScript/lessons/practice-app/
TEST_USER=admin
TEST_PASS=admin
// Load .env in cucumber.json using dotenv
{
  "default": {
    "requireModule": ["ts-node/register", "dotenv/config"],
    "require": ["src/steps/**/*.ts", "src/support/**/*.ts"],
    "paths": ["src/features/**/*.feature"]
  }
}
// Install dotenv
npm install --save-dev dotenv
