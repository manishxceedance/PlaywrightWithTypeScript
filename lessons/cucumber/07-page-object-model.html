Lesson 7: Page Object Model — Clean, Maintainable Automation

Level: Intermediate Organise your selectors and page actions into reusable classes to prevent duplication.

1. Why Page Object Model?

Without POM, selectors like '[data-testid="email"]' are scattered across dozens of step definition files. When a developer renames an element, you must update every file. With POM:

  • Each page has one class that owns all its selectors and actions.
  • Step definitions call page object methods — no selectors in step files.
  • When the UI changes, you update one class, not 20 step definitions.

2. Structure of a Page Object Class

// src/pages/LoginPage.ts
import { Page, expect } from '@playwright/test';

export class LoginPage {
  constructor(private page: Page) {}

  // Locators — private, only the class uses them
  private get emailInput()  { return this.page.locator('[data-testid="email"]'); }
  private get passwordInput() { return this.page.locator('[data-testid="password"]'); }
  private get submitButton()  { return this.page.locator('[data-testid="login-submit"]'); }
  private get errorMessage()  { return this.page.locator('[data-testid="login-error"]'); }

  // Actions — public, called by step definitions
  async navigate(baseUrl: string) {
    await this.page.goto(baseUrl + '/login');
  }

  async login(email: string, password: string) {
    await this.emailInput.fill(email);
    await this.passwordInput.fill(password);
    await this.submitButton.click();
  }

  // Assertions — public
  async assertErrorVisible(message: string) {
    await expect(this.errorMessage).toContainText(message);
  }
}

3. Adding Page Objects to the World

Instantiate page objects in the Before hook and store them on the World so all steps can use them: 

// src/support/world.ts
import { setWorldConstructor, World, IWorldOptions } from '@cucumber/cucumber';
import { Browser, BrowserContext, Page } from '@playwright/test';
import { LoginPage } from '../pages/LoginPage';
import { DashboardPage } from '../pages/DashboardPage';

export interface CustomWorld extends World {
  browser: Browser;
  context: BrowserContext;
  page: Page;
  baseUrl: string;
  loginPage: LoginPage;
  dashboardPage: DashboardPage;
}

class PlaywrightWorld extends World {
  browser!: Browser;
  context!: BrowserContext;
  page!: Page;
  baseUrl: string = process.env.BASE_URL ?? 'https://example.com';
  loginPage!: LoginPage;
  dashboardPage!: DashboardPage;

  constructor(options: IWorldOptions) {
    super(options);
  }
}

setWorldConstructor(PlaywrightWorld);
// src/support/hooks.ts  — add to the Before hook
Before(async function (this: CustomWorld) {
  this.context = await browser.newContext();
  this.page = await this.context.newPage();

  // Initialise page objects after page is created
  this.loginPage = new LoginPage(this.page);
  this.dashboardPage = new DashboardPage(this.page);
});

4. Step Definitions with Page Objects

Step definitions become thin wrappers — they delegate all browser interaction to the page object: 

// src/steps/login.steps.ts
import { Given, When, Then } from '@cucumber/cucumber';
import { CustomWorld } from '../support/world';

Given('the user is on the login page', async function (this: CustomWorld) {
  await this.loginPage.navigate(this.baseUrl);
});

When(
  'the user enters email {string} and password {string}',
  async function (this: CustomWorld, email: string, password: string) {
    await this.loginPage.login(email, password);
  }
);

Then(
  'an error message {string} should be displayed',
  async function (this: CustomWorld, message: string) {
    await this.loginPage.assertErrorVisible(message);
  }
);

5. DashboardPage Example

// src/pages/DashboardPage.ts
import { Page, expect } from '@playwright/test';

export class DashboardPage {
  constructor(private page: Page) {}

  private get welcomeHeading() { return this.page.locator('h1.welcome'); }
  private get userMenu()        { return this.page.locator('[data-testid="user-menu"]'); }
  private get logoutBtn()       { return this.page.locator('[data-testid="logout"]'); }

  async assertWelcomeContains(name: string) {
    await expect(this.welcomeHeading).toContainText(name);
  }

  async logout() {
    await this.userMenu.click();
    await this.logoutBtn.click();
    await this.page.waitForURL(/login/);
  }

  async isLoaded() {
    await expect(this.page).toHaveURL(/dashboard/);
    await expect(this.welcomeHeading).toBeVisible();
  }
}

6. Base Page Class (Optional but Useful)

For large projects, create a BasePage with shared behaviour like waiting for spinners to disappear: 

// src/pages/BasePage.ts
import { Page } from '@playwright/test';

export class BasePage {
  constructor(protected page: Page) {}

  async waitForLoadingToComplete() {
    await this.page.locator('.loading-spinner').waitFor({ state: 'hidden', timeout: 10000 });
  }

  async scrollToBottom() {
    await this.page.evaluate(() => window.scrollTo(0, document.body.scrollHeight));
  }
}

// src/pages/LoginPage.ts  — extends BasePage
import { BasePage } from './BasePage';

export class LoginPage extends BasePage {
  constructor(page: Page) { super(page); }

  async login(email: string, password: string) {
    await this.page.fill('[data-testid="email"]', email);
    await this.page.fill('[data-testid="password"]', password);
    await this.page.click('[data-testid="login-submit"]');
    await this.waitForLoadingToComplete();
  }
}

7. Recommended Folder Structure for POM

src/
  pages/
    BasePage.ts           ← Shared utility methods
    LoginPage.ts          ← Login page actions and assertions
    DashboardPage.ts      ← Dashboard page actions and assertions
    CheckoutPage.ts       ← Checkout flow
    SearchResultsPage.ts  ← Search results
  steps/
    login.steps.ts
    dashboard.steps.ts
    checkout.steps.ts
  support/
    world.ts              ← Holds page object references
    hooks.ts              ← Initialises page objects in Before
  features/
    login.feature
    checkout.feature

8. Anti-Patterns to Avoid

  • Selectors in step definitions — selectors belong only in page objects. If you have this.page.click('[data-testid="submit"]') in a step file, move it to a page object.
  • One massive page object — split by feature area: LoginPage, CheckoutPage, not one AllPagesEver.ts.
  • Business logic in page objects — page objects should navigate, fill, click, assert. Conditional logic and test data should stay in steps or test files.

Practice Exercise

  1. Create a LoginPage class with navigate, login, and assertError methods.
  2. Create a DashboardPage class with isLoaded and getUserName methods.
  3. Add both page objects to the World and initialise them in the Before hook.
  4. Refactor your login step definitions to use the page object methods instead of direct Playwright calls.
  5. Confirm all tests still pass after the refactor.
